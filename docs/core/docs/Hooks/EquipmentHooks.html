            return;
        }

    {
        Log.Debug("ItemPickupSystem.OnUpdate called");
        if (!Application.IsServer || __instance.__query_1414696066_0.IsEmpty) {
            return;
        }

        var entityManager = Global.EntityManager;
        var entities = __instance.__query_1414696066_0.ToEntityArray(Allocator.Temp);
        foreach (var entity in entities)
        {
            if (!entityManager.TryGetComponentData<EntityOwner>(entity, out var ownerData)) {
                continue;
            }
            
            var characterEntity = ownerData.Owner;
            if (!entityManager.TryGetComponentData<PlayerCharacter>(characterEntity, out var playerCharacter)) {
                continue;
            }
            
            Log.Debug("Publishing OnItemPickupEvent");
            EventBridge.PublishSafe(new OnItemPickupEvent(entity, new FromCharacter {
                Character = characterEntity,
                User = playerCharacter.UserEntity,
            }));
        }
        
        entities.Dispose();
    }
