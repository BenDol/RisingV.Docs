    [HarmonyPatch]
    public static class ModifyUnitStatBuffSystemPatch
    {
        private static readonly Logger Log = Logger.Create(nameof(ModifyUnitStatBuffSystemPatch), LogTags.Hooks,
            LogTags.Buffs);

        [HarmonyPrefix]
        [HarmonyPatch(typeof(ModifyUnitStatBuffSystem_Spawn), nameof(ModifyUnitStatBuffSystem_Spawn.OnUpdate))]
        private static void SpawnPrefix(ModifyUnitStatBuffSystem_Spawn __instance)
        {
            var entities = __instance.__query_35557666_0.ToEntityArray(Allocator.Temp);

            foreach (var entity in entities) {
                var prefabGuid = entity.GetAndLogPrefabGuid("[ModifyUnitStatBuffSystem.SpawnPrefix]"); entity.Read<PrefabGUID>();

                EventBridge.PublishSafe(new UnitStatBuffSpawnEvent(entity, prefabGuid));
            }
            
            entities.Dispose();
        }
    }

    [HarmonyPatch]
    public static class BuffDebugSystemPatch
    {
        private static readonly Logger Log = Logger.Create(nameof(BuffDebugSystemPatch), LogTags.Hooks,
            LogTags.Buffs);

        [HarmonyPrefix]
        [HarmonyPatch(typeof(BuffDebugSystem), nameof(BuffDebugSystem.OnUpdate))]
        private static void Prefix(BuffDebugSystem __instance)
        {
            var entities = __instance.__query_401358787_0.ToEntityArray(Allocator.Temp);
            foreach (var entity in entities) {
                var guid = entity.GetAndLogPrefabGuid("[BuffDebugSystem.Prefix]");

                var buffData = new BuffData(guid);

                var ownerEntity = __instance.EntityManager.GetComponentData<EntityOwner>(entity).Owner;
                if (__instance.EntityManager.TryGetComponentData(ownerEntity, out PlayerCharacter playerCharacter)) {
                    var userEntity = playerCharacter.UserEntity;
                    var userData = __instance.EntityManager.GetComponentData<User>(userEntity);
                    
                    EventBridge.PublishSafe(new PlayerBuffingEvent(userEntity, playerCharacter, userData, buffData));
                } else {
                    EventBridge.PublishSafe(new UnitBuffingEvent(entity, ownerEntity, buffData));
                }
            }
            
            entities.Dispose();
        }

        [HarmonyPostfix]
        [HarmonyPatch(typeof(BuffDebugSystem), nameof(BuffDebugSystem.OnUpdate))]
        private static void Postfix(BuffDebugSystem __instance)
        {
            var entities = __instance.__query_401358787_0.ToEntityArray(Allocator.Temp);
            foreach (var entity in entities) {
                var guid = entity.GetAndLogPrefabGuid("[BuffDebugSystem.Postfix]");
                
                var buffData = new BuffData(guid);
                
                var ownerEntity = __instance.EntityManager.GetComponentData<EntityOwner>(entity).Owner;
                if (__instance.EntityManager.TryGetComponentData(ownerEntity, out PlayerCharacter playerCharacter)) {
                    var userEntity = playerCharacter.UserEntity;
                    var userData = __instance.EntityManager.GetComponentData<User>(userEntity);
                    
                    EventBridge.PublishSafe(new PlayerBuffedEvent(userEntity, playerCharacter, userData, buffData));
                } else {
                    EventBridge.PublishSafe(new UnitBuffedEvent(entity, ownerEntity, buffData));
                }
            }
            
            entities.Dispose();
        }
    }
